Display Results

LW = 'LineWidth';
lw = 2;

subplot(311);
plot(r,'r',LW,lw);
hold on
plot(y,LW,lw)
hold off
title('Process Output')
ylabel('y');

subplot(312);
plot(d,LW,lw)
title('Disturbance')
ylabel('d');

subplot(313);
plot(u,LW,lw);
title('Control Input');
ylabel('u')
